	jQuery(document).ready(function() {

			$(".wrapper").attr("aria-hidden", true);

		/** errors.jsp
		 * Función que valida si se ha producido algun error en la jsp, y muestra el msg de error
		 * apropiado a el usuario.
		 */
		function checkErrors() {
			var hayError = false;
			var errorHtmlBody = "";
			
			var mostrarValidaSMS = checkContains();
			var mostrarModalErrorValidaSMS = checkContainsErrorValidaSMS();
			
			if (!mostrarValidaSMS &amp;&amp; !mostrarModalErrorValidaSMS){	
				
					
				if (hayError) {
					errorHtmlBody = "&lt;ul&gt;" + errorHtmlBody + "&lt;/ul&gt;";
					setModal("Se han encontrado los siguientes errores:", errorHtmlBody);
					jQuery(document).trigger("erroresControlados");
					jQuery("#modalGeneric").modal('show');
				}
			}
		}

		function checkHayError() {
			var hayError = false;
				
			
			return hayError;
		}
		
		/**
		 * Función que hace scroll hasta el primer campo con error (que contiene la clase campo_error)
		 */
		function scrollToError() {
			var primerError = jQuery(".campo_error:first");
			setTimeout(function() {
				jQuery(".campo_error:first").focus();
			}, 10);
			var scrollPane = jQuery("html");
			if (primerError &amp;&amp; primerError.offset()) {
				var scrollY = primerError.offset().top + scrollPane.scrollTop() - 50;
				scrollPane.animate({scrollTop : scrollY }, 500, 'swing');
			}
		}
		
		/**
		 * Función que valida si se ha producido algun mensaje en la jsp, y muestra el msg de error
		 * apropiado a el usuario.
		 */
		function checkAvisos() {
			var hayMensajes = false;
			var mensajeHtmlBody = "";
			
			
			
			if (hayMensajes) {
				mensajeHtmlBody = "&lt;ul&gt;" + mensajeHtmlBody + "&lt;/ul&gt;";
				setModal("Aviso", mensajeHtmlBody);
				jQuery("#modalGeneric").modal('show');
			}
		}
		
		function setModal (titulo, cuerpo, funcion, buttonText, buttonFunc) {
			jQuery("#myModalLabel").html(titulo);
			jQuery("#myModalBody").html(cuerpo);
			jQuery("button.close").click(funcion);
			
			if (buttonText &amp;&amp; buttonFunc){
				jQuery('#myModalFooter').removeClass("hidden");
				jQuery('#myModalButton').html(buttonText).click(buttonFunc);
			} else {
				jQuery('#myModalFooter').addClass("hidden");
			}
		}
		
		function addToModal (titulo, cuerpo) {
			jQuery("#myModalLabel").html(titulo);
			jQuery("#myModalBody").append(cuerpo);
		}
		 
		/**
		 * Muestra un msg de error pasado por parámetro.
		 */
		function setErrorsMessage(msg) {
			var arrayMsg = [];
			if (msg != null) {
				arrayMsg = msg.split("|");
			}
			var msgModified = "";
			for (var i = 0; i &lt; arrayMsg.length; i++) {
				msgModified += "&lt;li&gt;"
								+ arrayMsg[i] 
							+ "&lt;/li&gt;";
			}
			msgModified = "&lt;ul&gt;" + msgModified + "&lt;/ul&gt;";
